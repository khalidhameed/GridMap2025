<script

  src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"

  integrity="sha512-CJ+UZGkV3Y7AeGeDrZ3wqZRoR5X3EkXz19qDk+qDnKzH8OgPeNKKf4mjNBrZg3EvU8rgU5Rz1MQkzV7h12J4ow=="

  crossorigin=""

></script>


<script>

  // Initialize map

  const map = L.map('map').setView([51.2, 10.5], 6); // Centered on Germany


  // Add OpenStreetMap base layer

  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

    maxZoom: 19,

    attribution: '&copy; OpenStreetMap contributors'

  }).addTo(map);


  // Load GeoJSON data

  fetch('powerlines.geojson')

    .then(response => {

      if (!response.ok) {

        throw new Error("Could not load powerlines.geojson");

      }

      return response.json();

    })

    .then(data => {

      L.geoJSON(data, {

        style: feature => ({

          color:

            feature.properties.voltage === '380000' ? 'red' :

            feature.properties.voltage === '220000' ? 'orange' :

            'blue',

          weight: 2

        }),

        onEachFeature: (feature, layer) => {

          const name = feature.properties.name || 'Unnamed Line';

          layer.bindPopup(`<strong>${name}</strong>`);

        }

      }).addTo(map);

    })

    .catch(err => {

      console.error("Error loading GeoJSON:", err);

    });

</script>
